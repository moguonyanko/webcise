Object destructuring

let obj = {
    x : 100,
    y : 200,
    toString : () => {
        return obj.x + ", " + obj.y;
    }
};
let {x, y, toString} = obj;
return {x, y, toString};
let obj = {
    x : "foo",
    y : "bar",
    z : "baz"
